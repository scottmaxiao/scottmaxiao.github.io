##IOS-自定义tableViewCell     IOS提供的UITableViewCell类中包括一个UIImageView和两个UILabel，显示的时候可以设置四种布局方式。但是很多时候并不能满足我们的需求。我们常常会自定义tableViewCell内显示的内容。 #####**声明自定义tableViewCell** infoTableViewCell.h文件 

@interface infoTableViewCell : UITableViewCell

@property(nonatomic,strong)UILabel *namesLabel;
@property(nonatomic,strong)UITextField *valueTextField;
@end
infoTableViewCell.m文件 

@implementation infoTableViewCell

- (instancetype)initWithStyle:(UITableViewCellStyle)style reuseIdentifier:(NSString *)reuseIdentifier{
	self = [super initWithStyle:style reuseIdentifier:reuseIdentifier];
	if (self) {
	    _namesLabel=[[UILabel alloc] initWithFrame:CGRectZero];
		//设置label是否可以显示多行，0则显示多行
		_namesLabel.numberOfLines = 0;
		//设置单词折行方式
		_namesLabel.lineBreakMode = NSLineBreakByCharWrapping;
		_namesLabel.layer.borderColor=[UIColor redColor].CGColor;
		[self.contentView addSubview:_namesLabel];
		
		_valueTextField=[[UITextField alloc] initWithFrame:CGRectZero];
		[self.contentView addSubview:_valueTextField];
	}
	
	return self;
}

@end
#####**实现layoutSubviews方法** 

-(void)layoutSubviews{
    //必须调用父类的layoutSubviews方法，否则不会正常显示
	[super layoutSubviews];
	
	
	self.namesLabel.frame=...;
	self.namesLabel.font=...;
	self.namesLabel.text=...;
	
	self.valueTextField.frame=...;
	self.valueTextField.tag=...;
	self.valueTextField.text=...;
	self.valueTextField.font=...t;
	//键盘类型
	self.valueTextField.keyboardType=...;
	//键盘返回按钮类型
	self.valueTextField.returnKeyType=...;
	//自动大写设置
	self.valueTextField.autocapitalizationType=...;
	//自动更正设置
	self.valueTextField.autocorrectionType=...;
	//安全输入设置
	self.valueTextField.secureTextEntry=...;
	//水平对其方式
	self.valueTextField.textAlignment=...;
}
